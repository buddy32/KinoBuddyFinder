        let quizData = [];

        const usersAnswers = [];

        let userLocation = null;

        const userPool = [];


        function shuffleArray(array) {

            for (let i = array.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [array[i], array[j]] = [array[j], array[i]];

            }

        }


        function getUserLocation() {

            if (navigator.geolocation) {

                navigator.geolocation.getCurrentPosition(

                    (position) => {

                        userLocation = {

                            latitude: position.coords.latitude,

                            longitude: position.coords.longitude

                        };

                        console.log("User location retrieved:", userLocation);

                    },

                    (error) => {

                        console.error("Error retrieving location:", error);

                        alert("Standort konnte nicht abgerufen werden. Die Match-Funktion wird eingeschränkt.");

                    }

                );

            } else {

                alert("Geolocation wird von deinem Browser nicht unterstützt.");

            }

        }


        function displayQuestion(index) {

            const questionElement = document.getElementById('question');

            const answersElement = document.getElementById('answers');


            if (index >= quizData.length) {

                addToUserPool();

                return;

            }


            const { question, answers } = quizData[index];

            questionElement.textContent = question;


            answersElement.innerHTML = '';

            answers.forEach(answer => {

                const answerElement = document.createElement('div');

                answerElement.textContent = answer;

                answerElement.classList.add('answer');

                answerElement.onclick = () => {

                    usersAnswers.push(answer);

                    displayQuestion(index + 1);

                };

                answersElement.appendChild(answerElement);

            });

        }


        function addToUserPool() {

            const userData = {

                answers: usersAnswers,

                location: userLocation,

                timestamp: new Date()

            };

            userPool.push(userData);

            console.log("User added to pool:", userData);


            const matchedUser = findMatch(userData);

            if (matchedUser) {

                displayMatch();

            } else {

                alert("Kein KinoBuddy gefunden. Du bleibst im Pool und wirst bald gematcht!");

            }

        }


        function findMatch(newUser) {

            for (let user of userPool) {

                if (user !== newUser && user.answers.join() === newUser.answers.join()) {

                    return user;

                }

            }

            return null;

        }


        function displayMatch() {

            document.getElementById('quiz').style.display = 'none';

            document.getElementById('matchMessage').style.display = 'block';

            document.getElementById('chatBox').style.display = 'block';

        }


        document.getElementById('restart').onclick = () => {

            usersAnswers.length = 0;

            quizData = [];

            initializeQuiz();

            document.getElementById('quiz').style.display = 'block';

            document.getElementById('matchMessage').style.display = 'none';

            document.getElementById('chatBox').style.display = 'none';

            document.getElementById('chatMessages').innerHTML = '';

            displayQuestion(0);

        };


        const chatInput = document.getElementById('chatInput');

        const chatMessages = document.getElementById('chatMessages');


        chatInput.addEventListener('keypress', (e) => {

            if (e.key === 'Enter' && chatInput.value.trim() !== '') {

                const message = chatInput.value;

                const messageElement = document.createElement('div');

                messageElement.textContent = message;

                messageElement.classList.add('chat-message');

                chatMessages.appendChild(messageElement);

                chatInput.value = '';

            }

        });


        function initializeQuiz() {

            quizData = [...allQuizData];

            shuffleArray(quizData);

            quizData = quizData.slice(0, 10);

        }


        // Start the quiz

        getUserLocation();

        initializeQuiz();
